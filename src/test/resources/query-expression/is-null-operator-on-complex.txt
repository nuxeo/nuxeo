import module namespace extract = 'http://nuxeo.com/extract' at '/ext/nuxeo/extract.xqy';
let $paths :=("/document/ecm__id","/document/ecm__name","/document/ecm__parentId")
let $namespaces :=()
for $i in cts:search(
            fn:doc(),
            cts:not-query(
              cts:and-query((
                cts:element-query(
                  fn:QName("","info__info"),
                  cts:and-query(())
                ),
                cts:element-query(
                  fn:QName("","info__info"),
                  cts:element-query(
                    fn:QName("","width"),
                    cts:and-query(())
                  )
                )
              ))
            )
          )
    return extract:extract-nodes($i, $paths, $namespaces)
